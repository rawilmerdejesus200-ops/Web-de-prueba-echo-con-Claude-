// components/ContactSection.jsx
function useScrollReveal(ref) {
  const { useEffect, useState } = React;
  const [visible, setVisible] = useState(false);
  useEffect(() => {
    const obs = new IntersectionObserver(
      (entries) => entries.forEach(e => e.isIntersecting && setVisible(true)),
      { threshold: 0.15, rootMargin: '0px 0px -60px 0px' }
    );
    if (ref.current) obs.observe(ref.current);
    return () => obs.disconnect();
  }, []);
  return visible;
}

function ContactSection() {
  const { useRef } = React;
  const ref = useRef(null);
  const visible = useScrollReveal(ref);

  return (

      <div className="contact-ghost-text">ÉLITE</div>
      <div className={`contact-inner reveal ${visible ? 'reveal-visible' : ''}`}>

        <h2 className="contact-title">Begin Your<br /><em>Journey</em></h2>

          Our advisors work exclusively with a select clientele. Share your vision and we will curate an experience tailored to your desires.
        </p>
        <form className="contact-form" onSubmit={(e) => e.preventDefault()}>

            <input className="form-input" type="text" placeholder="Full Name" />
