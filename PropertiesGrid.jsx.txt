function FeatureIcon({ type }) {
  if (type === 'bed') {
    return <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M3 12l2-2m0 0l7-7 7 7M5 10v10a1 1 0 001 1h3m10-11l2 2m-2-2v10a1 1 0 01-1 1h-3m-6 0a1 1 0 001-1v-4a1 1 0 011-1h2a1 1 0 011 1v4a1 1 0 001 1m-6 0h6" /></svg>;
  }
  return <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><circle cx="12" cy="12" r="10" /><path d="M12 8v4l3 3" /></svg>;
}

function PropertyCard({ p, index }) {
  const { useEffect, useRef, useState } = React;
  const ref = useRef(null);
  const [visible, setVisible] = useState(false);

  useEffect(() => {
    const obs = new IntersectionObserver(
      (entries) => entries.forEach(e => e.isIntersecting && setVisible(true)),
      { threshold: 0.15, rootMargin: '0px 0px -60px 0px' }
    );
    if (ref.current) obs.observe(ref.current);
    return () => obs.disconnect();
  }, []);

  return (
    <div
      ref={ref}
      className={`property-card ${p.span} reveal ${visible ? 'reveal-visible' : ''}`}
      style={{ aspectRatio: p.ratio, transitionDelay: `${index * 0.12}s` }}
    >
      <img className="property-img" src={p.img} alt={p.name} loading="lazy" />
      <div className="property-overlay" />
      <div className="property-info">
        <div className="property-badge">{p.badge}</div>
        <div className="property-name">{p.name}</div>
        <div className="property-location">{p.location}</div>
        <div className="property-price">{p.price}</div>
        {p.features.length > 0 && (
          <div className="property-features">
            {p.features.map((f, i) => (
              <span key={i} className="property-feature"><FeatureIcon type={f.icon} />{f.label}</span>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}

}
