function HeroParticles() {
  const { useEffect, useRef } = React;
  const canvasRef = useRef(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const renderer = new THREE.WebGLRenderer({ canvas, alpha: true, antialias: true });
    renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
    renderer.setSize(window.innerWidth, window.innerHeight);

    const scene = new THREE.Scene();
    const camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 1000);
    camera.position.z = 5;

    const count = 160;
    const geometry = new THREE.BufferGeometry();
    const positions = new Float32Array(count * 3);
    const speeds = new Float32Array(count);
    for (let i = 0; i < count; i++) {
      positions[i * 3] = (Math.random() - 0.5) * 18;
      positions[i * 3 + 1] = (Math.random() - 0.5) * 12;
      positions[i * 3 + 2] = (Math.random() - 0.5) * 6;
      speeds[i] = Math.random() * 0.003 + 0.001;
    }
    geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));

    const material = new THREE.PointsMaterial({
      color: 0xc9a84c,
      size: 0.04,
      transparent: true,
      opacity: 0.6,
      sizeAttenuation: true,
      blending: THREE.AdditiveBlending,
      depthWrite: false,
    });
    const particles = new THREE.Points(geometry, material);
    scene.add(particles);

    const rings = [];
    for (let i = 0; i < 3; i++) {
      const ringGeo = new THREE.TorusGeometry(1.4 + i * 0.9, 0.005, 8, 80);
      const ringMat = new THREE.MeshBasicMaterial({ color: 0xc9a84c, transparent: true, opacity: 0.07 - i * 0.015 });
      const ring = new THREE.Mesh(ringGeo, ringMat);
      ring.rotation.x = Math.PI / 4 + i * 0.3;
      ring.position.set(2.5, -1, -3 + i * 0.5);
      scene.add(ring);
      rings.push(ring);
    }

    let raf, t = 0;
    function animate() {
      raf = requestAnimationFrame(animate);
      t += 0.008;
      const pos = geometry.attributes.position.array;
      for (let i = 0; i < count; i++) {
        pos[i * 3 + 1] += speeds[i];
        if (pos[i * 3 + 1] > 7) pos[i * 3 + 1] = -7;
      }
      geometry.attributes.position.needsUpdate = true;
      particles.rotation.y = t * 0.05;
      rings.forEach((r, i) => { r.rotation.z = t * 0.08 * (i % 2 === 0 ? 1 : -1); });
      renderer.render(scene, camera);
    }
    animate();

    const onResize = () => {
      renderer.setSize(window.innerWidth, window.innerHeight);
      camera.aspect = window.innerWidth / window.innerHeight;
      camera.updateProjectionMatrix();
    };
    window.addEventListener('resize', onResize);
    return () => {
      cancelAnimationFrame(raf);
      window.removeEventListener('resize', onResize);
      renderer.dispose();
    };
  }, []);

  return <canvas ref={canvasRef} className="hero-particles-canvas" />;
}

function StatCounter({ target, prefix = '', suffix = '', decimals = 0, triggerKey }) {
  const { useEffect, useState } = React;
  const [value, setValue] = useState(0);

  useEffect(() => {
    if (!triggerKey) return;
    let start = 0;
    const step = target / 50;
    const id = setInterval(() => {
      start += step;
      if (start >= target) { start = target; clearInterval(id); }
      setValue(start);
    }, 16);
    return () => clearInterval(id);
  }, [triggerKey]);

  return <>{prefix}{decimals ? value.toFixed(decimals) : Math.round(value)}{suffix}</>;
}

function Hero({ ready, videoUrl }) {
  const { useEffect, useRef } = React;
  const videoRef = useRef(null);

  useEffect(() => {
    if (!ready) return;
    const tl = gsap.timeline({ delay: 0.2 });
    gsap.set(
      ['.hero-eyebrow', '.hero-title', '.hero-desc', '.hero-ctas', '.hero-scroll', '.hero-stat'],
      { y: 40, opacity: 0 }
    );
    tl.to('.hero-eyebrow', { opacity: 1, y: 0, duration: 0.9, ease: 'power3.out' })
      .to('.hero-title', { opacity: 1, y: 0, duration: 1, ease: 'power3.out' }, '-=0.5')
      .to('.hero-desc', { opacity: 1, y: 0, duration: 0.8, ease: 'power3.out' }, '-=0.6')
      .to('.hero-ctas', { opacity: 1, y: 0, duration: 0.8, ease: 'power3.out' }, '-=0.5')
      .to('.hero-scroll', { opacity: 1, y: 0, duration: 0.6, ease: 'power3.out' }, '-=0.4')
      .to('.hero-stat', { opacity: 1, y: 0, duration: 0.6, stagger: 0.1, ease: 'power3.out' }, '-=0.3');

    // Parallax on hero video
    const onScroll = () => {
      const scrolled = window.scrollY;
      if (videoRef.current && scrolled < window.innerHeight) {
        videoRef.current.style.transform = `translateY(${scrolled * 0.3}px)`;
      }
    };
    window.addEventListener('scroll', onScroll);
    return () => window.removeEventListener('scroll', onScroll);
  }, [ready]);

  return (
    <section id="hero" className="hero">
      <video ref={videoRef} className="hero-video" autoPlay muted loop playsInline>
        <source src={videoUrl} type="video/mp4" />
      </video>
      <div className="hero-overlay" />
      <div className="hero-overlay-bottom" />
      <HeroParticles />

      <div className="hero-content">
        <div className="hero-eyebrow">
          <div className="hero-eyebrow-line" />
          <span>Exclusive Collection 2025</span>
        </div>
        <h1 className="hero-title">
          Where<br /><em>Architecture</em><br />Meets<br />Velocity
        </h1>
        <p className="hero-desc">
          Curated luxury properties and supercars for the discerning few.
          Every residence tells a story. Every drive, a statement.
        </p>
        <div className="hero-ctas">
          <a href="#properties" className="btn-primary cursor-hover">
            Explore Properties
            <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M7 17L17 7M17 7H7M17 7v10" /></svg>
          </a>
          <a href="#supercars" className="btn-ghost cursor-hover">View Supercars</a>
        </div>
      </div>

      <div className="hero-scroll">
        <span>Scroll</span>
        <div className="scroll-line" />
      </div>

      <div className="hero-stats">
        <div className="hero-stat">
          <div className="num"><StatCounter target={148} triggerKey={ready} /></div>
          <div className="label">Properties</div>
        </div>
        <div className="hero-stat">
          <div className="num">$<StatCounter target={2.4} decimals={1} triggerKey={ready} />B</div>
          <div className="label">Portfolio Value</div>
        </div>
        <div className="hero-stat">
          <div className="num"><StatCounter target={34} triggerKey={ready} /></div>
          <div className="label">Countries</div>
        </div>
      </div>
    </section>
  );
}
