function GarageScroller() {
  const { useEffect, useRef } = React;
  const containerRef = useRef(null);
  const trackRef = useRef(null);
  const dragState = useRef({ isDragging: false, startX: 0, scrollLeft: 0 });

  useEffect(() => {
    const container = containerRef.current;
    const track = trackRef.current;
    if (!container || !track) return;

    container.style.cursor = 'grab';

    const onMouseDown = (e) => {
      dragState.current.isDragging = true;
      dragState.current.startX = e.pageX - container.offsetLeft;
      dragState.current.scrollLeft = container.scrollLeft;
      container.style.cursor = 'grabbing';
    };
    const onMouseMove = (e) => {
      if (!dragState.current.isDragging) return;
      const x = e.pageX - container.offsetLeft;
      container.scrollLeft = dragState.current.scrollLeft - (x - dragState.current.startX);
    };
    const onMouseUp = () => { dragState.current.isDragging = false; container.style.cursor = 'grab'; };

    let touchStartX = 0, touchScrollLeft = 0;
    const onTouchStart = (e) => {
      touchStartX = e.touches[0].pageX;
      touchScrollLeft = container.scrollLeft;
    };
    const onTouchMove = (e) => {
      const diff = touchStartX - e.touches[0].pageX;
      container.scrollLeft = touchScrollLeft + diff;
    };

    container.addEventListener('mousedown', onMouseDown);
    container.addEventListener('mousemove', onMouseMove);
    container.addEventListener('mouseup', onMouseUp);
    container.addEventListener('mouseleave', onMouseUp);
    container.addEventListener('touchstart', onTouchStart, { passive: true });
    container.addEventListener('touchmove', onTouchMove, { passive: true });

    // GSAP scroll-jacked horizontal movement tied to vertical scroll
    gsap.registerPlugin(ScrollTrigger);
    const st = gsap.to(track, {
      x: () => -(track.scrollWidth - window.innerWidth - 96),
      ease: 'none',
      scrollTrigger: {
        trigger: '#supercars',
        start: 'top center',
        end: 'bottom top',
        scrub: 1.5,
      },
    });

    return () => {
      container.removeEventListener('mousedown', onMouseDown);
      container.removeEventListener('mousemove', onMouseMove);
      container.removeEventListener('mouseup', onMouseUp);
      container.removeEventListener('mouseleave', onMouseUp);
      container.removeEventListener('touchstart', onTouchStart);
      container.removeEventListener('touchmove', onTouchMove);
      if (st.scrollTrigger) st.scrollTrigger.kill();
    };
  }, []);

  return (
    <section id="supercars" className="garage-section">
      <div className="garage-header">
        <div className="section-eyebrow">Curated Fleet</div>
        <h2 className="section-title">The <em>Garage</em><br />Collection</h2>
      </div>
      <div className="cars-drag-hint">Drag to explore</div>
      <div ref={containerRef} className="cars-scroll-container">
        <div ref={trackRef} className="cars-track">
          {CARS.map(c => <CarCard key={c.model} c={c} />)}
        </div>
      </div>
    </section>
  );
}
